# Data Cleaning EDA

# Replace Categorical missing values with mode
dfall['ship-city'].fillna(dfall['ship-city'].mode()[0],inplace=True)

dfall['ship-state'].fillna(dfall['ship-state'].mode()[0],inplace=True)

dfall['ship-postal-code'].fillna(dfall['ship-postal-code'].mode()[0],inplace=True)
dfall['ship-postal-code']=dfall['ship-postal-code'].astype(int)

dfall['promotion-ids'].fillna(dfall['promotion-ids'].mode()[0],inplace=True)


# Clean the junk data in the states
dfall['ship-state']=dfall['ship-state'].str.strip()
dfall['ship-state']=dfall['ship-state'].str.upper()
dfall['ship-state']=dfall['ship-state'].replace({
    'RAJSTHAN':'RAJASTHAN','RJ':'RAJASTHAN','RAJSHTHAN':'RAJASTHAN','PONDICHERRY':'PUDUCHERRY','NL':'NAGALAND','AR':'ARUNACHAL PRADESH',
    'ORISSA':'ODISHA','PUNJAB/MOHALI/ZIRAKPUR':'PUNJAB','PB':'PUNJAB','NEW DELHI':'DELHI'
})

# Remove those data which are invalid amounts as this was causing a very low R2 score
dfall=dfall[dfall['Amount'].notnull()]
dfall=dfall[dfall['Amount']>0]

# Add new columns for day, month & year based on Order date
def split_date_column(df, date_column):
    df[date_column]=pd.to_datetime(df[date_column], format='%m-%d-%y')
    df['day']=df[date_column].dt.day
    df['month']=df[date_column].dt.month
    df['year']=df[date_column].dt.year
    return df
dfall = split_date_column(dfall,'Date')


# Reduce the number of groups for Size feature. Adding this feature increased the R2 score
dfall['Size']=dfall['Size'].replace({'3XL':'XL','6XL':'XL','XXL':'XL','5XL':'XL','4XL':'XL'})
dfall['Size']=dfall['Size'].replace({'XS':'S'})
